    private UserContext GetDbContext() // to Get Entity framework database context
        {
            var options = new DbContextOptionsBuilder<UserContext>()
                .UseInMemoryDatabase(databaseName: Guid.NewGuid().ToString()) // Unique name per test
                .Options;

            return new UserContext(options); // this will create new dbcontext in memory
        }

        [Fact] // indicates test methods
        private async Task IsEmailIdUnique_shouldCheckUniqueEmail_ReturnBoolean()
        {
            // ARRANGE
            var context = GetDbContext();
            var user = new ShoppingUserModel {Name="rahul kumar", Email = "rahul@gmail.com", Password = "abc", Phone = "98767"};
            context.ShoppingUser.Add(user);
            context.SaveChanges();
            var testEmail = "pankaj@gmail.com";
            var service = new UserService(context, null, null);


            // ACT
            CancellationToken ct = new CancellationToken();
            var result1 = await service.IsEmailIdUnique(testEmail, ct);
            var result2 = await service.IsEmailIdUnique("rahul@gmail.com", ct);

            //ASSERT
            // result.Should().NotB
            result1.Should().Be(true);
            result2.Should().Be(false);
        }
